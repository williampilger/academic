int LDR = A5;
int LED = 12;

void setup() {
	pinMode(LED, OUTPUT);
  	Serial.begin(9600);
  digitalWrite(LED, HIGH);
}

int leitura = 0;

void loop () {
  leitura = analogRead(LDR);
  
  if(leitura >= 800) digitalWrite(LED, HIGH);
  else digitalWrite(LED, LOW);
  
  Serial.println(leitura);
  delay(100);
}
